package euler004

const max = 999
const min = 100

func Solve() int {
	i := max
	candidate := 0
	for {
		j := i
		for {
			x := i * j
			if reversible(x) {
				if candidate < x {
					candidate = x
				}
			}
			j--
			if j < min {
				break
			}
		}
		i--
		if i < min {
			break
		}
	}
	return candidate
}

func reversible(arg int) bool {
	x := arg
	if x%10 == 0 {
		return false
	}
	y := 0

	for {
		y += (x % 10)
		x = x / 10
		if x == 0 {
			break
		}
		y = y * 10
	}

	return arg == y
}
